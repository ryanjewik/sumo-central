powershell:
docker exec -it kafka kafka-topics.sh --create --bootstrap-server localhost:9092 --topic sumo.webhooks --partitions 6 --replication-factor 1

docker exec -it kafka kafka-topics.sh --create --bootstrap-server localhost:9092 --topic sumo.api --partitions 6 --replication-factor 1

bash:
docker exec -it kafka kafka-topics.sh \
  --create \
  --bootstrap-server localhost:9092 \
  --topic sumo.webhooks \
  --partitions 6 \
  --replication-factor 1

docker exec -it kafka kafka-topics.sh \
  --create \
  --bootstrap-server localhost:9092 \
  --topic sumo.api \
  --partitions 6 \
  --replication-factor 1

point both apps to kafka:
$env:KAFKA_BROKERS = "localhost:9092"

make sure common is importable:
$env:PYTHONPATH = "$PWD;$env:PYTHONPATH"
